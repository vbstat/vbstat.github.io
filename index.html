Moment

											
## ex-2.3
import numpy as np
x=[16,40,12,46,47,69,32,59,52,49,63]
r=3 #r=1,2,3
print(sum(np.power(x,r))/len(x))  #Raw moment
print(sum(np.power(x-(np.sum(x)/len(x)),r))/len(x))  #central moment
